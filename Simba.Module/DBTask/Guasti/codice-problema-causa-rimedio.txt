 private XPCollection<ProblemaCausa> fListaApparatoProblemaCausa;
        [PersistentAlias("fListaApparatoProblemaCausa"), System.ComponentModel.DisplayName("Elenco Problema/Causa")]
        //[Appearance("RdL.fListaApparatoProblemaCausa", Enabled = false)]
        [MemberDesignTimeVisibility(false)]
        public XPCollection<ProblemaCausa> ListaApparatoProblemaCausas
        {
            get
            {
                if (StdApparato == null)
                {
                    return null;
                }
                if (fListaApparatoProblemaCausa == null && StdApparato != null)
                {
                    fListaApparatoProblemaCausa = new XPCollection<ProblemaCausa>(Session);
                    RefreshListaFiltrataProblemiCausa();
                }
                fListaApparatoProblemaCausa.Select(pp => pp.Cause);
                return fListaApparatoProblemaCausa;
            }
        }

        private void RefreshListaFiltrataProblemiCausa()
        {
            if (fListaApparatoProblema == null)
            {
                return;
            }
            if (fListaApparatoProblemaCausa == null)
            {
                return;
            }
            if (StdApparato == null)
            {
                return;
            }
            if (Problema == null)
            {
                return;
            }
            //if (Sistema == null) And Sistema.Oid == {1}  , Sistema.Oid.ToString()
            //{
            //    return;
            //}
            //Problema.ProblemaCausas
            var ParCriteria = string.Format("StdApparato.Oid = {0}", StdApparato.Oid.ToString());
            fListaApparatoProblema.Criteria = CriteriaOperator.Parse(ParCriteria);
            OnChanged("ListaFiltraApparatoProblemas");
        }





		  private XPCollection<ApparatoProblema> fListaApparatoProblema;
        [PersistentAlias("fListaApparatoProblema"), System.ComponentModel.DisplayName("Elenco Problema")]
        // [Appearance("RdL.ListaApparatoProblemas", Enabled = false)]
        [MemberDesignTimeVisibility(false)]
        public XPCollection<ApparatoProblema> ListaFiltraApparatoProblemas
        {
            get
            {
                if (fListaFiltrataRdLs == null && Impianto != null)
                {
                    fListaApparatoProblema = new XPCollection<ApparatoProblema>(Session);
                    RefreshListaFiltrataProblemi();
                }
                return fListaApparatoProblema;
            }
        }
        private void RefreshListaFiltrataProblemi()
        {
            if (fListaApparatoProblema == null)
            {
                return;
            }
            if (StdApparato == null)
            {
                return;
            }
            var ParCriteria = string.Format("StdApparato.Oid = {0} ", StdApparato.Oid.ToString());

            fListaApparatoProblema.Criteria = CriteriaOperator.Parse(ParCriteria);
            OnChanged("ListaFiltraApparatoProblemas");
        }
