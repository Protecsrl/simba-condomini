create or replace view v_causa_rimedio as
select 'CA' || std.oid || '-' || pr.oid || '-' || ca.oid as CAUSA,
       rim.descrizione as Name,
       rim.descrizione,
       rim.oid as RIMEDIO,
       'RIM' || std.oid || '-' || pr.oid || '-' || ca.oid || '-' || rim.oid as oid

from PCRPROBLEMI pr, PCRCAUSE ca, PCRPROBCAUSA prca, PCRRIMEDI rim, PCRCAUSARIMEDIO carim,
     PCRAPPPROBLEMA ap, APPARATOSTD std 
where std.oid=ap.stdapparato and ap.pcrproblemi=pr.oid and prca.pcrcause=ca.oid and prca.pcrappproblema=pr.oid and carim.pcrrimedi=rim.oid 
      and carim.pcrprobcausa=ca.oid;
