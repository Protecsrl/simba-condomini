create or replace view v_clusteredifici_mansione as
select x.clusteredifici || '-' || x.mansioni as codice,
       x.clusteredifici,
     --  x.descrizione,
        x.mansioni,
      sum( x.carico) carico
  from (
         select distinct clust.oid as clusteredifici,
                        --imp.descrizione,
                        edi.oid as Edificio,
                imp.oid as Impianto,
                a.oid as Apparato,
                sk.mansioniopt as mansioni,
                ask.schedamp,
               -- f.descrizione,f.cadenze_anno,
/*                 (a.quantita * sk.tempoopt * sk.numman *
                              f.cadenze_anno)  carico1,
                (1 +
                 ((Nvl(ku.valore, 1) - 1) + (Nvl(a.kguasto, 1) - 1) +
                 (Nvl(kd.valore, 1) - 1) + (Nvl(kc.valore, 1) - 1) +
                 (Nvl(kb.valore, 1) - 1) \*+ Nvl(c.presidio, 0)*\
                 )) carico2,*/
                (a.quantita * sk.tempoopt * sk.numman * f.cadenze_anno) *
                (1 +
                 ((Nvl(ku.valore, 1) - 1) + (Nvl(a.kguasto, 1) - 1) +
                 (Nvl(kd.valore, 1) - 1) + (Nvl(kc.valore, 1) - 1) +
                 (Nvl(kb.valore, 1) - 1) /*+ Nvl(c.presidio, 0)*/
                 )) carico
  from clusteredifici  clust,
       edificio         edi,
       impianto         imp,
       apparato         a,
       APPARATOSCHEDAMP ask,
       schedemp         sk,
       frequenze        f,
       kdimensione kd,
       kutenza  ku,
       kcondizione kc,
       kubicazione kb

 where clust.oid = edi.clusteredifici
  and   edi.oid = imp.edificio
 and   imp.oid = a.impianto
   and a.oid = ask.apparato
   and ask.schedamp = sk.oid
   and sk.frequenzaopt = f.oid
   and sk.kdimensione= kd.oid (+)
   and sk.kcondizione= kc.oid (+)
   and sk.kutenza= ku.oid (+)
  and  sk.kubicazione = kb.oid (+)
   and sk.mansioniopt is not null
   --and imp.oid = 206

        ) x
 --where x.impianto = 206
 group by x.clusteredifici,  x.mansioni--,--x.descrizione
 order by clusteredifici, carico
;
